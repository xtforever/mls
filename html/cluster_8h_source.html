54/* We have 16384 hash slots. The hash slot of a given key is obtained
55 * as the least significant 14 bits of the crc16 of the key.
56 *
57 * However, if the key contains the {...} pattern, only the part between
58 * { and } is hashed. This may be useful in the future to force certain
59 * keys to be in the same node (assuming no resharding is in progress). */
60static inline unsigned int keyHashSlot(const char *key, int keylen) {
61 int s, e; /* start-end indexes of { and } */
62
63 for (s = 0; s < keylen; s++)
64 if (key[s] == '{') break;
65
66 /* No '{' ? Hash the whole key. This is the base case. */
67 if (likely(s == keylen)) return crc16(key,keylen) & 0x3FFF;
68
69 /* '{' found? Check if we have the corresponding '}'. */
70 for (e = s+1; e < keylen; e++)
71 if (key[e] == '}') break;
72
73 /* No '}' or nothing between {} ? Hash the whole key. */
74 if (e == keylen || e == s+1) return crc16(key,keylen) & 0x3FFF;
75
76 /* If we are here there is both a { and a } on its right. Hash
77 * what is in the middle between { and }. */
78 return crc16(key+s+1,e-s-1) & 0x3FFF;
79}

225/* API for alternative cluster implementations to start and coordinate
226 * Atomic Slot Migration (ASM).
227 *
228 * These two functions drive ASM for alternative cluster implementations.
229 * - clusterAsmProcess(...) impl -> redis: initiates/advances/cancels ASM operations
230 * - clusterAsmOnEvent(...) redis -> impl: notifies state changes
231 *
232 * Generic steps for an alternative implementation:
233 * - On destination side, implementation calls clusterAsmProcess(ASM_EVENT_IMPORT_START)
234 * to start an import operation.
235 * - Redis calls clusterAsmOnEvent() when an ASM event occurs.
236 * - On the source side, Redis will call clusterAsmOnEvent(ASM_EVENT_HANDOFF_PREP)
237 * when slots are ready to be handed off and the write pause is needed.
238 * - Implementation stops the traffic to the slots and calls clusterAsmProcess(ASM_EVENT_HANDOFF)
239 * - On the destination side, Redis calls clusterAsmOnEvent(ASM_EVENT_TAKEOVER)
240 * when destination node is ready to take over the slot, waiting for ownership change.
241 * - Cluster implementation updates the config and calls clusterAsmProcess(ASM_EVENT_DONE)
242 * to notify Redis that the slots ownership has changed.
243 *
244 * Sequence diagram for import:
245 * - Note: shows only the events that cluster implementation needs to react.
246 *
247 * ┌───────────────┐ ┌───────────────┐ ┌───────────────┐ ┌───────────────┐
248 * │ Destination │ │ Destination │ │ Source │ │ Source │
249 * │ Cluster impl │ │ Master │ │ Master │ │ Cluster impl │
250 * └───────┬───────┘ └───────┬───────┘ └───────┬───────┘ └───────┬───────┘
251 * │ │ │ │
252 * │ ASM_EVENT_IMPORT_START │ │ │
253 * ├─────────────────────────────►│ │ │
254 * │ │ CLUSTER SYNCSLOTS <arg> │ │
255 * │ ├────────────────────────►│ │
256 * │ │ │ │
257 * │ │ SNAPSHOT(restore cmds) │ │
258 * │ │◄────────────────────────┤ │
259 * │ │ Repl stream │ │
260 * │ │◄────────────────────────┤ │
261 * │ │ │ ASM_EVENT_HANDOFF_PREP │
262 * │ │ ├────────────────────────────►│
263 * │ │ │ ASM_EVENT_HANDOFF │
264 * │ │ │◄────────────────────────────┤
265 * │ │ Drain repl stream │ │
266 * │ │◄────────────────────────┤ │
267 * │ ASM_EVENT_TAKEOVER │ │ │
268 * │◄─────────────────────────────┤ │ │
269 * │ │ │ │
270 * │ ASM_EVENT_DONE │ │ │
271 * ├─────────────────────────────►│ │ ASM_EVENT_DONE │
272 * │ │ │◄────────────────────────────┤
273 * │ │ │ │
274 */
275
276#define ASM_EVENT_IMPORT_START 1 /* Start a new import operation (destination side) */
277#define ASM_EVENT_CANCEL 2 /* Cancel an ongoing import/migrate operation (source and destination side) */
278#define ASM_EVENT_HANDOFF_PREP 3 /* Slot is ready to be handed off to the destination shard (source side) */
279#define ASM_EVENT_HANDOFF 4 /* Notify that the slot can be handed off (source side) */
280#define ASM_EVENT_TAKEOVER 5 /* Ready to take over the slot, waiting for config change (destination side) */
281#define ASM_EVENT_DONE 6 /* Notify that import/migrate is completed, config is updated (source and destination side) */
282
283#define ASM_EVENT_IMPORT_PREP 7 /* Import is about to start, the implementation may reject by returning C_ERR */
284#define ASM_EVENT_IMPORT_STARTED 8 /* Import started */
285#define ASM_EVENT_IMPORT_FAILED 9 /* Import failed */
286#define ASM_EVENT_IMPORT_COMPLETED 10 /* Import completed (config updated) */
287#define ASM_EVENT_MIGRATE_PREP 11 /* Migrate is about to start, the implementation may reject by returning C_ERR */
288#define ASM_EVENT_MIGRATE_STARTED 12 /* Migrate started */
289#define ASM_EVENT_MIGRATE_FAILED 13 /* Migrate failed */

293/* Called by cluster implementation to request an ASM operation. (cluster impl --> redis)
294 * Valid values for 'event':
295 * ASM_EVENT_IMPORT_START
296 * ASM_EVENT_CANCEL
297 * ASM_EVENT_HANDOFF
298 * ASM_EVENT_DONE
299 *
300 * For ASM_EVENT_IMPORT_START, 'task_id' should be a unique string.
301 * For other events (ASM_EVENT_CANCEL, ASM_EVENT_HANDOFF, ASM_EVENT_DONE),
302 * 'task_id' should match the ID from the corresponding import operation.
303 * Usage:
304 * char *task_id = malloc(CLUSTER_NAMELEN + 1);
305 * getRandomHexChars(task_id, CLUSTER_NAMELEN);
306 * task_id[CLUSTER_NAMELEN] = '\0';
307 *
308 * slotRangeArray *slots = slotRangeArrayCreate(1);
309 * slotRangeArraySet(slots, 0, 0, 1000);
310 *
311 * const char *err = NULL;
312 * int ret = clusterAsmProcess(task_id, ASM_EVENT_IMPORT_START, slots, &err);
313 * zfree(task_id);
314 * slotRangeArrayFree(slots);
315 *
316 * if (ret != C_OK) {
317 * perror(err);
318 * return;
319 * }
320 *
321 * For ASM_EVENT_CANCEL, if `task_id` is NULL, all tasks will be cancelled.
322 * If `arg` parameter is provided, it should be a pointer to an int. It will be
323 * set to the number of tasks cancelled.
324 *
325 * Return value:
326 * - Returns C_OK on success, C_ERR on failure and 'err' will be set to the
327 * error message.
328 *
329 * Memory management:
330 * - There is no ownership transfer of 'task_id', 'err' or `slotRangeArray`.
331 * - `task_id` and `slotRangeArray` should be allocated and be freed by the
332 * caller. Redis internally will make a copy of these.
333 * - `err` is allocated by Redis and should NOT be freed by the caller.
334 **/
335int clusterAsmProcess(const char *task_id, int event, void *arg, char **err);

337/* Called when an ASM event occurs to notify the cluster implementation. (redis --> cluster impl)
338 *
339 * `arg` will point to a `slotRangeArray` for the following events:
340 * ASM_EVENT_IMPORT_PREP
341 * ASM_EVENT_IMPORT_STARTED
342 * ASM_EVENT_MIGRATE_PREP
343 * ASM_EVENT_MIGRATE_STARTED
344 * ASM_EVENT_HANDOFF_PREP
345 *
346 * Memory management:
347 * - Redis owns the `task_id` and `slotRangeArray`.
348 *
349 * Returns C_OK on success.
350 *
351 * If the cluster implementation returns C_ERR for ASM_EVENT_IMPORT_PREP or
352 * ASM_EVENT_MIGRATE_PREP, operation will not start.
353 **/
354int clusterAsmOnEvent(const char *task_id, int event, void *arg);
