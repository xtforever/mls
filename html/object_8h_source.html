1/*
2 * Redis objects overview
3 *
4 * robj (struct redisObject) is the fundamental in-memory container that can hold
5 * values of different logical types (strings, lists, sets, hashes, sorted sets,
6 * streams, modules, ...). It contains:
7 * - type: one of OBJ_STRING, OBJ_LIST, OBJ_SET, OBJ_ZSET, OBJ_HASH, OBJ_STREAM,
8 * OBJ_MODULE, ...
9 * - encoding: an implementation detail of how the value is represented in
10 * memory for the given type (see OBJ_ENCODING_* below). For example,
11 * strings may be RAW/EMBSTR/INT, sets may be INTSET or HT, etc.
12 * - lru: either LRU information (relative to the global LRU clock) or LFU data
13 * when LFU is enabled.
14 * - refcount: reference counting for object lifetime management.
15 * - ptr: a pointer to the underlying value payload (SDS, dict, quicklist, ...).
16 *
17 * Object encodings
18 * -----------------
19 * Some kinds of objects (strings, hashes, sets, zsets, lists) have multiple
20 * possible in-memory encodings optimized for memory footprint and speed. The
21 * 'encoding' field indicates which representation is currently used (see
22 * OBJ_ENCODING_* defines below).
23 *
24 * kvobj (key-value object)
25 * ------------------------
26 * kvobj is a specific use of robj that additionally embeds the key (and optional
27 * metadata) alongside the value. It is identified by the iskvobj flag in robj.
28 * The distinction is mostly declarative for clarity and may be enforced later with
29 * explicit casting in places. Conceptually, robj and kvobj are in a relation
30 * similar to that of a parent-child class hierarchy.
31 *
32 * When iskvobj is set, it also contains:
33 * - metabits: bitmap of additional metadata attached to the object.
34 * - lru: LRU time (relative to global lru_clock) or LFU data (see robj above).
35 * - embedded key: the key string is stored inline after the struct.
36 * - embedded value: for small strings, the value is stored inline after the key.
37 *
38 * Example layout with key and embedded value "myvalue":
39 * +--------------+--------------+--------------------+----------------------+
40 * | serverObject | key-hdr-size | sdshdr5 "mykey" \0 | sdshdr8 "myvalue" \0 |
41 * | 16 bytes | 1 byte | 1 + 5 + 1 | 3 + 7 + 1 |
42 * +--------------+--------------+--------------------+----------------------+
43 *
44 * kvobj with metadata (+expiration)
45 * ---------------------------------
46 * Up to 8 metadata classes are supported, each storing one 8-byte field.
47 * Class 0 is reserved for expiration time. Metadata blocks are placed before
48 * the kvobj itself, in reverse class order.
49 *
50 * Example of a key with expiration time (metabits=0b00000001):
51 * +--------------+--------------+--------------+--------------------+
52 * | Expiry Time | serverObject | key-hdr-size | sdshdr5 "mykey" \0 |
53 * | 8 byte | 16 bytes | 1 byte | 1 + 5 + 1 |
54 * +--------------+--------------+--------------+--------------------+
55 * ^
56 * +---- kvobjCreate() returns pointer here
57 *
58 * Example with metadata of class1 and class3 attached (metabits=0b00001010):
59 * +--------------+--------------+--------------+--------------+--------------------+
60 * | meta (class3)| meta (class1)| serverObject | key-hdr-size | sdshdr5 "mykey" \0 |
61 * | 8 byte | 8 byte | 16 bytes | 1 byte | 1 + 5 + 1 |
62 * +--------------+--------------+--------------+--------------+--------------------+
63 * ^
64 * +---- kvobjCreate() returns pointer here
65 *
66 */

99struct redisObject {
100 unsigned type:4;
101 unsigned encoding:4;
102 unsigned refcount : OBJ_REFCOUNT_BITS;
103 unsigned iskvobj : 1; /* 1 if this struct serves as a kvobj base */
104
105 /* metabits and lru are Relevant only when iskvobj is set: */
106 unsigned metabits :8; /* Bitmap of metadata (+expiry) attached to this kvobj */
107 unsigned lru:LRU_BITS; /* LRU time (relative to global lru_clock) or
108 * LFU data (least significant 8 bits frequency
109 * and most significant 16 bits access time). */
110 void *ptr;
111};
