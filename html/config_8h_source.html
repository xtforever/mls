254/* Sometimes after including an OS-specific header that defines the
255 * endianness we end with __BYTE_ORDER but not with BYTE_ORDER that is what
256 * the Redis code uses. In this case let's define everything without the
257 * underscores. */
258#ifndef BYTE_ORDER
259#ifdef __BYTE_ORDER
260#if defined(__LITTLE_ENDIAN) && defined(__BIG_ENDIAN)
261#ifndef LITTLE_ENDIAN
262#define LITTLE_ENDIAN __LITTLE_ENDIAN
263#endif
264#ifndef BIG_ENDIAN
265#define BIG_ENDIAN __BIG_ENDIAN
266#endif
267#if (__BYTE_ORDER == __LITTLE_ENDIAN)
268#define BYTE_ORDER LITTLE_ENDIAN
269#else
270#define BYTE_ORDER BIG_ENDIAN
271#endif
272#endif
273#endif
274#endif
