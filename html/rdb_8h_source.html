23/* Defines related to the dump file format. To store 32 bits lengths for short
24 * keys requires a lot of space, so we check the most significant 2 bits of
25 * the first byte to interpreter the length:
26 *
27 * 00|XXXXXX => if the two MSB are 00 the len is the 6 bits of this byte
28 * 01|XXXXXX XXXXXXXX => 01, the len is 14 bits, 6 bits + 8 bits of next byte
29 * 10|000000 [32 bit integer] => A full 32 bit len in net byte order will follow
30 * 10|000001 [64 bit integer] => A full 64 bit len in net byte order will follow
31 * 11|OBKIND this means: specially encoded object will follow. The six bits
32 * number specify the kind of object that follows.
33 * See the RDB_ENC_* defines.
34 *
35 * Lengths up to 63 are stored using a single byte, most DB keys, and may
36 * values, will fit inside. */
37#define RDB_6BITLEN 0
38#define RDB_14BITLEN 1
39#define RDB_32BITLEN 0x80
40#define RDB_64BITLEN 0x81
41#define RDB_ENCVAL 3
42#define RDB_LENERR UINT64_MAX
43
44/* When a length of a string object stored on disk has the first two bits
45 * set, the remaining six bits specify a special encoding for the object
46 * accordingly to the following defines: */
47#define RDB_ENC_INT8 0 /* 8 bit signed integer */
48#define RDB_ENC_INT16 1 /* 16 bit signed integer */
49#define RDB_ENC_INT32 2 /* 32 bit signed integer */
50#define RDB_ENC_LZF 3 /* string compressed with FASTLZ */
