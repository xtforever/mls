21#define AE_NONE 0 /* No events registered. */
22#define AE_READABLE 1 /* Fire when descriptor is readable. */
23#define AE_WRITABLE 2 /* Fire when descriptor is writable. */
24#define AE_BARRIER 4 /* With WRITABLE, never fire the event if the
25 READABLE event already fired in the same event
26 loop iteration. Useful when you want to persist
27 things to disk before sending replies, and want
28 to do that in a group fashion. */
