26/* Note: sdshdr5 is never used, we just access the flags byte directly.
27 * However is here to document the layout of type 5 SDS strings. */
28struct __attribute__ ((__packed__)) sdshdr5 {
29 unsigned char flags; /* 3 lsb of type, and 5 msb of string length */
30 char buf[];
31};
32struct __attribute__ ((__packed__)) sdshdr8 {
33 uint8_t len; /* used */
34 uint8_t alloc; /* excluding the header and null terminator */
35 unsigned char flags; /* 3 lsb of type, 5 unused bits */
36 char buf[];
37};
38struct __attribute__ ((__packed__)) sdshdr16 {
39 uint16_t len; /* used */
40 uint16_t alloc; /* excluding the header and null terminator */
41 unsigned char flags; /* 3 lsb of type, 5 unused bits */
42 char buf[];
43};
44struct __attribute__ ((__packed__)) sdshdr32 {
45 uint32_t len; /* used */
46 uint32_t alloc; /* excluding the header and null terminator */
47 unsigned char flags; /* 3 lsb of type, 5 unused bits */

57#define SDS_TYPE_5 0
58#define SDS_TYPE_8 1
59#define SDS_TYPE_16 2
60#define SDS_TYPE_32 3
61#define SDS_TYPE_64 4
62#define SDS_TYPE_MASK 7
63#define SDS_TYPE_BITS 3
64#define SDS_HDR_VAR(T,s) struct sdshdr##T *sh = (void*)((s)-(sizeof(struct sdshdr##T)));
65#define SDS_HDR(T,s) ((struct sdshdr##T *)((s)-(sizeof(struct sdshdr##T))))
66#define SDS_TYPE_5_LEN(s) (((unsigned char)(s[-1])) >> SDS_TYPE_BITS)

68static inline unsigned char sdsType(sds s) {
69 unsigned char flags = s[-1];
70 return flags & SDS_TYPE_MASK;
71}

73/* Returns a user data bit stored in the SDS header by sdsSetAuxBit. The bit
74 * index is 0-4. Returns 0 or 1. Always returns 0 for SDS_TYPE_5. */
75static inline int sdsGetAuxBit(sds s, int bit) {
76 if (sdsType(s) == SDS_TYPE_5)
77 return 0;
78
79 unsigned char flags = s[-1];
80 return (flags & (1U << (SDS_TYPE_BITS + bit))) != 0U;
81}
82
83/* Stores a bit in an unused area in the SDS header, except for SDS_TYPE_5. The
84 * bit index is 0-4. The value is 0 or 1. The aux bits are lost if the SDS is
85 * auto-resized. This is only for special uses like immutable SDS embedded in
86 * other structures. */
87static inline void sdsSetAuxBit(sds s, int bit, int value) {
88 if (sdsType(s) == SDS_TYPE_5) return;
89 unsigned char flags = s[-1];
90 if (value) {
91 flags |= 1U << (SDS_TYPE_BITS + bit);
92 } else {
93 flags &= ~(1U << (SDS_TYPE_BITS + bit));
94 }
95 s[-1] = (char)flags;
96}

98static inline size_t sdslen(const sds s) {
99 switch (sdsType(s)) {
100 case SDS_TYPE_5: return SDS_TYPE_5_LEN(s);
101 case SDS_TYPE_8:
102 return SDS_HDR(8,s)->len;
103 case SDS_TYPE_16:
104 return SDS_HDR(16,s)->len;
105 case SDS_TYPE_32:
106 return SDS_HDR(32,s)->len;
107 case SDS_TYPE_64:
108 return SDS_HDR(64,s)->len;
109 }
110 return 0;
111}

113static inline size_t sdsavail(const sds s) {
114 switch(sdsType(s)) {
115 case SDS_TYPE_5: {
116 return 0;
117 }

123 SDS_HDR_VAR(16,s);

130 case SDS_TYPE_64: {
131 SDS_HDR_VAR(64,s);
132 return sh->alloc - sh->len;
133 }
134 }
135 return 0;

138static inline void sdssetlen(sds s, size_t newlen) {
139 switch(sdsType(s)) {
140 case SDS_TYPE_5:

142 unsigned char *fp = ((unsigned char*)s)-1;
143 *fp = SDS_TYPE_5 | (newlen << SDS_TYPE_BITS);

185/* Return the total size of the allocation of the specified sds string,
186 * including:
187 * 1) The sds header before the pointer.
188 * 2) The string.
189 * 3) The free buffer at the end if any.
190 * 4) The implicit null term.
191 */
192static inline size_t sdsAllocSize(sds s) {
193 switch(sdsType(s)) {
194 case SDS_TYPE_5:
195 return sizeof(struct sdshdr5) + SDS_TYPE_5_LEN(s) + 1;
196 case SDS_TYPE_8:
197 return sizeof(struct sdshdr8) + SDS_HDR(8,s)->alloc + 1;
198 case SDS_TYPE_16:
199 return sizeof(struct sdshdr16) + SDS_HDR(16,s)->alloc + 1;
200 case SDS_TYPE_32:
201 return sizeof(struct sdshdr32) + SDS_HDR(32,s)->alloc + 1;
202 case SDS_TYPE_64:
203 return sizeof(struct sdshdr64) + SDS_HDR(64,s)->alloc + 1;
204 }
205 return 0;
206}
