1/*
2 * Key Metadata (keymeta)
3 *
4 * High-level idea
5 * ----------------
6 * keymeta is a framework for attaching & maintaining metadata to keys.
7 *
8 * - Up to 8 different metadata classes can be registered globally. First one
9 * is reserved for EXPIRE.
10 * - Each class has a unique ID and name, like modules datat-types names, yet
11 * it has its own namespace.
12 * - Each key metadata class provides a set of callbacks for key lifecycle operations,
13 * ensuring consistent handling across copy, rename, logical removal (unlink),
14 * actual deallocation (free), persistence (RDB/AOF), and defragmentation.
15 * - Each key can carry up to 8 independent metadata values. Each value is related
16 * to a specific metadata class.
17 * - The 8-byte slot can either hold inline data or a pointer/handle to a larger,
18 * externally managed structure.
19 *
20 * Relation to other components
21 * ----------------------------
22 * - kvobj: 8 bits metabits field in kvobj is used to indicate active metadata.
23 * bit number corresponds to class ID.
24 * - Expiration: class ID 0 is reserved for TTL/expire;
25 * - Registration: redisServer.keyMetaClass[] stores registered classes. Modules
26 * register via keyMetaClassCreate (see redismodule.h) and may provide callbacks
27 * for persistence, copy/rename behavior, and lifecycle hooks (unlink/free).
28 * - modules: modules can register metadata classes and provide callbacks.
29 */

102/* KeyMetaSpec - Used by dbAddInternal() to describe metadata of a new key */
103typedef struct KeyMetaSpec {
104 uint16_t numMeta; /* Num active metadata entries. Aligned with metabits */
105 uint16_t metabits;
106
107 /* Array of metadata values. Entries are populated in reverse order
108 * (from the end of the array backward) to make bulk copying with
109 * memcpy more efficient. During insertion, the next slot is:
110 * meta[KEY_META_ID_MAX - (++numMeta)]
111 *
112 * For example if numMeta=2, and metabits=0b101, then the last entry holds
113 * value for class 0, and the previous entry holds value for class 2.
114 */
115 uint64_t meta[KEY_META_ID_MAX];
116} KeyMetaSpec;
