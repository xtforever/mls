128/* The connection module does not deal with listening and accepting sockets,
129 * so we assume we have a socket when an incoming connection is created.
130 *
131 * The fd supplied should therefore be associated with an already accept()ed
132 * socket.
133 *
134 * connAccept() may directly call accept_handler(), or return and call it
135 * at a later time. This behavior is a bit awkward but aims to reduce the need
136 * to wait for the next event loop, if no additional handshake is required.
137 *
138 * IMPORTANT: accept_handler may decide to close the connection, calling connClose().
139 * To make this safe, the connection is only marked with CONN_FLAG_CLOSE_SCHEDULED
140 * in this case, and connAccept() returns with an error.
141 *
142 * connAccept() callers must always check the return value and on error (C_ERR)
143 * a connClose() must be called.
144 */
145
146static inline int connAccept(connection *conn, ConnectionCallbackFunc accept_handler) {
147 return conn->type->accept(conn, accept_handler);
148}
149
150/* Establish a connection. The connect_handler will be called when the connection
151 * is established, or if an error has occurred.
152 *
153 * The connection handler will be responsible to set up any read/write handlers
154 * as needed.
155 *
156 * If C_ERR is returned, the operation failed and the connection handler shall
157 * not be expected.
158 */
159static inline int connConnect(connection *conn, const char *addr, int port, const char *src_addr,
160 ConnectionCallbackFunc connect_handler) {
161 return conn->type->connect(conn, addr, port, src_addr, connect_handler);
162}

174/* Write to connection, behaves the same as write(2).
175 *
176 * Like write(2), a short write is possible. A -1 return indicates an error.
177 *
178 * The caller should NOT rely on errno. Testing for an EAGAIN-like condition, use
179 * connGetState() to see if the connection state is still CONN_STATE_CONNECTED.
180 */
181static inline int connWrite(connection *conn, const void *data, size_t data_len) {
182 return conn->type->write(conn, data, data_len);
183}

185/* Gather output data from the iovcnt buffers specified by the members of the iov
186 * array: iov[0], iov[1], ..., iov[iovcnt-1] and write to connection, behaves the same as writev(3).
187 *
188 * Like writev(3), a short write is possible. A -1 return indicates an error.
189 *
190 * The caller should NOT rely on errno. Testing for an EAGAIN-like condition, use
191 * connGetState() to see if the connection state is still CONN_STATE_CONNECTED.
192 */
193static inline int connWritev(connection *conn, const struct iovec *iov, int iovcnt) {
194 return conn->type->writev(conn, iov, iovcnt);
195}
196
197/* Read from the connection, behaves the same as read(2).
198 *
199 * Like read(2), a short read is possible. A return value of 0 will indicate the
200 * connection was closed, and -1 will indicate an error.
201 *
202 * The caller should NOT rely on errno. Testing for an EAGAIN-like condition, use
203 * connGetState() to see if the connection state is still CONN_STATE_CONNECTED.
204 */
205static inline int connRead(connection *conn, void *buf, size_t buf_len) {
206 int ret = conn->type->read(conn, buf, buf_len);
207 return ret;
208}

279/* Format an IP,port pair into something easy to parse. If IP is IPv6
280 * (matches for ":"), the ip is surrounded by []. IP and port are just
281 * separated by colons. This the standard to display addresses within Redis. */
282static inline int formatAddr(char *buf, size_t buf_len, char *ip, int port) {
283 return snprintf(buf, buf_len, strchr(ip,':') ?
284 "[%s]:%d" : "%s:%d", ip, port);
285}
