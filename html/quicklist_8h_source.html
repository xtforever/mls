38/* quicklistNode is a 32 byte struct describing a listpack for a quicklist.
39 * We use bit fields keep the quicklistNode at 32 bytes.
40 * count: 16 bits, max 65536 (max lp bytes is 65k, so max count actually < 32k).
41 * encoding: 2 bits, RAW=1, LZF=2.
42 * container: 2 bits, PLAIN=1 (a single item as char array), PACKED=2 (listpack with multiple items).
43 * recompress: 1 bit, bool, true if node is temporary decompressed for usage.
44 * attempted_compress: 1 bit, boolean, used for verifying during testing.
45 * dont_compress: 1 bit, boolean, used for preventing compression of entry.
46 * extra: 9 bits, free for future use; pads out the remainder of 32 bits */
47typedef struct quicklistNode {
48 struct quicklistNode *prev;
49 struct quicklistNode *next;
50 unsigned char *entry;
51 size_t sz; /* entry size in bytes */
52 unsigned int count : 16; /* count of items in listpack */
53 unsigned int encoding : 2; /* RAW==1 or LZF==2 */
54 unsigned int container : 2; /* PLAIN==1 or PACKED==2 */
55 unsigned int recompress : 1; /* was this node previous compressed? */
56 unsigned int attempted_compress : 1; /* node can't compress; too small */
57 unsigned int dont_compress : 1; /* prevent compression of entry that will be used later */
58 unsigned int extra : 9; /* more bits to steal for future usage */
59} quicklistNode;
60
61/* quicklistLZF is a 8+N byte struct holding 'sz' followed by 'compressed'.
62 * 'sz' is byte length of 'compressed' field.
63 * 'compressed' is LZF data with total (compressed) length 'sz'
64 * NOTE: uncompressed length is stored in quicklistNode->sz.
65 * When quicklistNode->entry is compressed, node->entry points to a quicklistLZF */
66typedef struct quicklistLZF {
67 size_t sz; /* LZF size in bytes*/
68 char compressed[];
69} quicklistLZF;

71/* Bookmarks are padded with realloc at the end of the quicklist struct.
72 * They should only be used for very big lists if thousands of nodes were the
73 * excess memory usage is negligible, and there's a real need to iterate on them
74 * in portions.
75 * When not used, they don't add any memory overhead, but when used and then
76 * deleted, some overhead remains (to avoid resonance).
77 * The number of bookmarks used should be kept to minimum since it also adds
78 * overhead on node deletion (searching for a bookmark to update). */
79typedef struct quicklistBookmark {
80 quicklistNode *node;
81 char *name;
82} quicklistBookmark;

99/* quicklist is a 40 byte struct (on 64-bit systems) describing a quicklist.
100 * 'count' is the number of total entries.
101 * 'len' is the number of quicklist nodes.
102 * 'compress' is: 0 if compression disabled, otherwise it's the number
103 * of quicklistNodes to leave uncompressed at ends of quicklist.
104 * 'fill' is the user-requested (or default) fill factor.
105 * 'bookmarks are an optional feature that is used by realloc this struct,
106 * so that they don't consume memory when not used. */
107typedef struct quicklist {
108 quicklistNode *head;
109 quicklistNode *tail;
110 unsigned long count; /* total count of all entries in all listpacks */
111 unsigned long len; /* number of quicklistNodes */
112 size_t alloc_size; /* total allocated memory (in bytes) */
113 signed int fill : QL_FILL_BITS; /* fill factor for individual nodes */
114 unsigned int compress : QL_COMP_BITS; /* depth of end nodes not to compress;0=off */
115 unsigned int bookmark_count: QL_BM_BITS;
116 quicklistBookmark bookmarks[];
117} quicklist;
