9/* -----------------------------------------------------------------------------
10 * Entry
11 * -----------------------------------------------------------------------------
12 * An Entry represents a packed field-value pair with optional expiration metadata.
13 * Currently it is used only by Hash.
14 *
15 * There are 3 different formats for the "entry". In all cases, the "entry"
16 * pointer points into the allocation and is identical to the "field" sds pointer.
17 *
18 * Type 1: Field sds type is an SDS_TYPE_5
19 * With this type, both the key and value are embedded in the entry. Expiration
20 * is not allowed as the SDS_TYPE_5 (on field) doesn't contain any aux bits
21 * to encode the existence of an expiration. Extra padding is included in
22 * the value to the size of the physical block.
23 *
24 * entry
25 * |
26 * +---------V------------+----------------------------+
27 * | Field | Value |
28 * | sdshdr5 | "foo" \0 | sdshdr8 "bar" \0 (padding) |
29 * +---------+------------+----------------------------+
30 *
31 * Identified by: field sds type is SDS_TYPE_5
32 *
33 *
34 * Type 2: Field sds type is an SDS_TYPE_8 type
35 * With this type, both the key and value are embedded. Extra bits in the
36 * sdshdr8 (on field) are used to encode aux flags which may indicate the
37 * presence of an optional expiration. Extra padding is included in the value
38 * to the size of the physical block.
39 *
40 * entry
41 * |
42 * +--------------+---------V------------+----------------------------+
43 * | Expire (opt) | Field | Value |
44 * | ExpireMeta | sdshdr8 | "foo" \0 | sdshdr8 "bar" \0 (padding) |
45 * +--------------+---------+------------+----------------------------+
46 *
47 * Identified by: sds type is SDS_TYPE_8 AND has embedded value
48 *
49 *
50 * Type 3: Value is an sds, referenced by pointer
51 * With this type, the key is embedded, and the value is an sds, referenced
52 * by pointer. Extra bits in the sdshdr8/16/32 (on field) are used to encode
53 * aux flags which indicate the presence of a value by pointer. An aux bit
54 * may indicate the presence of an optional expiration. Note that the
55 * "field" is not padded, so there's no direct way to identify the length of the allocation.
56 *
57 * entry
58 * |
59 * +--------------+---------------+----------V----------+--------+
60 * | Expire (opt) | Value | Field | / / / /|
61 * | ExpireMeta | sds (pointer) | sdshdr8+ | "foo" \0 |/ / / / |
62 * +--------------+-------+-------+----------+----------+--------+
63 * |
64 * +-> sds value
65 *
66 * Identified by: Aux bit FIELD_SDS_AUX_BIT_ENTRY_HAS_VALUE_PTR
67 */

99/* Creates a new entry with the given field, value, and optional expiry.
100 * Flags can be ENTRY_TAKE_VALUE (take ownership of value if not embedded) and
101 * ENTRY_HAS_EXPIRY (entry has expiration metadata).
102 * If usable is not NULL, it will be set to the actual allocated size. */
103Entry *entryCreate(sds field, sds value, uint32_t flags, size_t *usable);

105/* Updates the value and/or expiry of an existing entry.

107 * Flags can be ENTRY_TAKE_VALUE (take ownership of value if not embedded) and
108 * ENTRY_HAS_EXPIRY (reserve space for existing expiry).
109 * If usableDiff not NULL, it will be set to diff in mem usage (newUsable - oldUsable) */
110Entry *entryUpdate(Entry *entry, sds value, uint32_t flags, ssize_t *usableDiff);
