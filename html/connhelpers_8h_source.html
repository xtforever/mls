16/* These are helper functions that are common to different connection
17 * implementations (currently sockets in connection.c and TLS in tls.c).
18 *
19 * Currently helpers implement the mechanisms for invoking connection
20 * handlers and tracking connection references, to allow safe destruction
21 * of connections from within a handler.
22 */
23
24/* Increment connection references.
25 *
26 * Inside a connection handler, we guarantee refs >= 1 so it is always
27 * safe to connClose().
28 *
29 * In other cases where we don't want to prematurely lose the connection,
30 * it can go beyond 1 as well; currently it is only done by connAccept().
31 */
32static inline void connIncrRefs(connection *conn) {
33 conn->refs++;
34}
35
36/* Decrement connection references.
37 *
38 * Note that this is not intended to provide any automatic free logic!
39 * callHandler() takes care of that for the common flows, and anywhere an
40 * explicit connIncrRefs() is used, the caller is expected to take care of
41 * that.
42 */
43
44static inline void connDecrRefs(connection *conn) {
45 conn->refs--;
46}
47
48static inline int connHasRefs(connection *conn) {
49 return conn->refs;
50}
51
52/* Helper for connection implementations to call handlers:
53 * 1. Increment refs to protect the connection.
54 * 2. Execute the handler (if set).
55 * 3. Decrement refs and perform deferred close, if refs==0.
56 */
57static inline int callHandler(connection *conn, ConnectionCallbackFunc handler) {
58 connIncrRefs(conn);
59 if (handler) handler(conn);
60 connDecrRefs(conn);
61 if (conn->flags & CONN_FLAG_CLOSE_SCHEDULED) {
62 if (!connHasRefs(conn)) connClose(conn);
63 return 0;
64 }
65 return 1;
66}
