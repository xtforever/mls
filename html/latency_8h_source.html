49/* Start monitoring an event. We just set the current time. */
50#define latencyStartMonitor(var) if (server.latency_monitor_threshold) { \
51 var = mstime(); \
52} else { \
53 var = 0; \
54}
55
56/* End monitoring an event, compute the difference with the current time
57 * to check the amount of time elapsed. */
58#define latencyEndMonitor(var) if (server.latency_monitor_threshold) { \
59 var = mstime() - var; \
60}
61
62/* Add the sample only if the elapsed time is >= to the configured threshold. */
63#define latencyAddSampleIfNeeded(event,var) \
64 if (server.latency_monitor_threshold && \
65 (var) >= server.latency_monitor_threshold) \
66 latencyAddSample((event),(var));
67
68/* Remove time from a nested event. */
69#define latencyRemoveNestedEvent(event_var,nested_var) \
70 event_var += nested_var;
