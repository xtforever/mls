6 * To make it easier to map each part of the format string with each argument,
7 * this file provides a way to write
8 *
9 * printf("a = %s, b = %s, c = %s\n",
10 * arg1, arg2, arg3);
11 *
12 * as
13 *
14 * printf(FMTARGS("a = %s, ", arg1,
15 * "b = %s, ", arg2,
16 * "c = %s\n", arg3));
17 *
18 * FMTARGS is variadic macro which is implemented by passing on its arguments to
19 * two other variadic macros of which one extracts the odd (the formats) and the
20 * other extracts the even (the arguments). The definitions of these macros
21 * include counting the number of macro arguments. Therefore, they don't accept
22 * an unlimited number of arguments. Currently it is fixed to a maximum of 120
23 * formats and arguments.
24 */
25#ifndef FMTARGS_H
26#define FMTARGS_H

28/* A macro to count the number of arguments. */
29#define NARG(...) NARG_I(__VA_ARGS__,RSEQ_N())
30#define NARG_I(...) ARG_N(__VA_ARGS__)

32/* Define a macro which will call an arbitrary macro appended with a number indicating
33 * the number of arguments it has. */
34#define VFUNC_N_(name, n) name##n
35#define VFUNC_N(name, n) VFUNC_N_(name, n)
36#define VFUNC(func, ...) VFUNC_N(func, NARG(__VA_ARGS__)) (__VA_ARGS__)

38/* Macros to extract the formats and the arguments from the fmt-arg pairs and
39 * then combine them again with all formats first and the arguments last. */
40#define COMPACT_FMT(...) VFUNC(COMPACT_FMT_, __VA_ARGS__)
41#define COMPACT_VALUES(...) VFUNC(COMPACT_VALUES_, __VA_ARGS__)
42#define FMTARGS(...) COMPACT_FMT(__VA_ARGS__), COMPACT_VALUES(__VA_ARGS__)
