13/*
14 * Script.c unit provides an API for functions and eval
15 * to interact with Redis. Interaction includes mostly
16 * executing commands, but also functionalities like calling
17 * Redis back on long scripts or check if the script was killed.
18 *
19 * The interaction is done using a scriptRunCtx object that
20 * need to be created by the user and initialized using scriptPrepareForRun.
21 *
22 * Detailed list of functionalities expose by the unit:
23 * 1. Calling commands (including all the validation checks such as
24 * acl, cluster, read only run, ...)
25 * 2. Set Resp
26 * 3. Set Replication method (AOF/REPLICATION/NONE)
27 * 4. Call Redis back to on long running scripts to allow Redis reply
28 * to clients and perform script kill
29 */
30
31/*
32 * scriptInterrupt function will return one of those value,
33 *
34 * - SCRIPT_KILL - kill the current running script.
35 * - SCRIPT_CONTINUE - keep running the current script.
36 */
37#define SCRIPT_KILL 1
38#define SCRIPT_CONTINUE 2
