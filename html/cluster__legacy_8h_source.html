89/* Message types.
90 *
91 * Note that the PING, PONG and MEET messages are actually the same exact
92 * kind of packet. PONG is the reply to ping, in the exact format as a PING,
93 * while MEET is a special PING that forces the receiver to add the sender
94 * as a node (if it is not already in the list). */
95#define CLUSTERMSG_TYPE_PING 0 /* Ping */
96#define CLUSTERMSG_TYPE_PONG 1 /* Pong (reply to Ping) */
97#define CLUSTERMSG_TYPE_MEET 2 /* Meet "let's join" message */
98#define CLUSTERMSG_TYPE_FAIL 3 /* Mark node xxx as failing */
99#define CLUSTERMSG_TYPE_PUBLISH 4 /* Pub/Sub Publish propagation */
100#define CLUSTERMSG_TYPE_FAILOVER_AUTH_REQUEST 5 /* May I failover? */
101#define CLUSTERMSG_TYPE_FAILOVER_AUTH_ACK 6 /* Yes, you have my vote */
102#define CLUSTERMSG_TYPE_UPDATE 7 /* Another node slots configuration */

259/* clusterMsg defines the gossip wire protocol exchanged among Redis cluster
260 * members, which can be running different versions of redis-server bits,
261 * especially during cluster rolling upgrades.
262 *
263 * Therefore, fields in this struct should remain at the same offset from
264 * release to release. The static asserts below ensure that incompatible
265 * changes in clusterMsg are caught at compile time.
266 */

273static_assert(offsetof(clusterMsg, count) == 14, "unexpected field offset");
